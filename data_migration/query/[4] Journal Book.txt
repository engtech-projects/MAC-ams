TRUNCATE TABLE journal_book;
INSERT INTO journal_book ( book_code, book_name, book_src, book_ref, book_head, book_flag ) (
	SELECT
		BOOK_CODE AS "book_code",
		BOOK_NAME AS "book_name",
		SOURCE_PRE AS "book_src",
		BOOK_REF AS "book_ref",
		BOOK_HEAD AS "book_head",
		BOOK_FLAG AS "book_flag"	 
	FROM
	glbook 
	)
